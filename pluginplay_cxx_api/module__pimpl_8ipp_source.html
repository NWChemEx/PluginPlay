33inline auto ModulePIMPL::not_set(const type::input_map& in_inputs) const {
34 assert_mod_();
35 not_set_type probs;
36
37 // This is all of the not ready inputs
38 auto in_probs = not_set_guts_(m_inputs_);
39
40 // Now pull out those set by the property type
41 for(const auto& [k, v] : in_inputs)
42 if(in_probs.count(k)) in_probs.erase(k);
43 if(!in_probs.empty()) probs.emplace("Inputs", std::move(in_probs));
44
45 auto submod_probs = not_set_guts_(m_submods_);
46 if(!submod_probs.empty())
47 probs.emplace("Submodules", std::move(submod_probs));
48 return probs;
49}

101inline bool ModulePIMPL::is_memoizable() const {
102 assert_mod_();

136inline std::string ModulePIMPL::profile_info() const {
137 std::stringstream ss;
138 ss << m_timer_;
139 std::string tab(" ");

151inline auto ModulePIMPL::run(type::input_map ps) {
152 auto time_now = time_stamp();
153 m_timer_.reset();
154 assert_mod_();
155 // Check the inputs we were just given
156 for(const auto& [k, v] : ps)
157 if(!v.ready()) throw std::runtime_error("Inputs are not ready");
158
159 // Merge with bound and see if we are ready
160 if(!ready(ps)) {
161 // Make a dummy module with this PIMPL so we can print out why it's not
162 // ready.
163 Module dummy(std::make_unique<ModulePIMPL>(*this));
164 throw std::runtime_error(print_not_ready(dummy, ps));
165 }
166
167 lock();
168
169 ps = merge_inputs_(ps);
170
171 if(is_memoizable() && m_cache_ && m_cache_->count(ps)) {
172 m_timer_.record(time_now);
173 return m_cache_->uncache(ps);
174 }
175
176 // not there so run
177 auto rv = m_base_->run(ps, m_submods_);
178
179 if(!m_cache_ || !is_memoizable()) {
180 m_timer_.record(time_now);
181 return rv;
182 }
183
184 // cache result
185 m_cache_->cache(ps, std::move(rv));
186 m_timer_.record(time_now);
187 return m_cache_->uncache(ps);
188}

190inline bool ModulePIMPL::operator==(const ModulePIMPL& rhs) const {
191 if(has_module() != rhs.has_module()) return false;
192 if(locked() != rhs.locked()) return false;
193 if(!has_module()) return true;
194
195 if(std::tie(inputs(), submods(), property_types()) !=
196 std::tie(rhs.inputs(), rhs.submods(), rhs.property_types()))
197 return false;
198 return (*m_base_ == *rhs.m_base_);
199}

211inline typename ModulePIMPL::submod_uuid_map ModulePIMPL::submod_uuids() const {
212 submod_uuid_map rv;
213 for(const auto& [k, v] : submods()) {
214 // Prepend the submodule key to each of its submodules' keys
215 // N.B. This essentially scopes the sub submodule keys
216 for(const auto& [sub_k, sub_v] : v.submod_uuids()) {
217 rv.emplace(k + ":" + sub_k, sub_v);
218 }
219 rv.emplace(k, v.uuid());
220 }
221 return rv;
222}

224inline type::input_map ModulePIMPL::merge_inputs_(
225 type::input_map in_inputs) const {
226 for(const auto& [k, v] : m_inputs_)
227 if(!in_inputs.count(k)) in_inputs[k] = v;
228
229 // TODO: It probably makes sense to create an Input class which tracks this
230 // and allows using submods as inputs
231 std::string submod_key = "__PLUGIN_PLAY__ SUBMOD KEYS __PLUGIN_PLAY__";
232 ModuleInput temp;
233 temp.set_type<submod_uuid_map>();
234 temp.change(submod_uuids());
235 in_inputs.emplace(submod_key, temp);
236 return in_inputs;
237}

246 std::set<type::key> probs;
247 for(const auto& [k, v] : map)
