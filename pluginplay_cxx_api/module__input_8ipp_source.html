30template<typename T>
31T ModuleInput::value() {
32 // Did the user ask for the value back by copy?
33 constexpr bool by_value = std::is_same_v<std::decay_t<T>, T>;
34
35 // Want a const reference or a copy? Defer to ModuleInput::value() const
36 if constexpr(detail_::IsConstRef<T>::value || by_value)
37 return const_cast<const ModuleInput&>(*this).value<T>();
38
39 // Know user wants it by read/write reference (okay if we copied the value)
40
41 // Check how it's stored. If not a copy (i.e., it's a const ref) we can't
42 // return read/write ref
43 if(m_is_cref_)
44 throw std::runtime_error("Input is a const ref, so you can't get it "
45 "back as a modifiable reference.");
46
47 return any::any_cast<T>(get_());
48}

95template<typename T>
96auto& ModuleInput::set_type() {
97 // Make sure the user isn't trying to set the input type to read/write
98 // reference.
99 constexpr bool is_ref = std::is_reference_v<T>;
100 constexpr bool is_c_ref = detail_::IsConstRef<T>::value;
101 static_assert(is_c_ref || !is_ref,
102 "T should not be an lvalue or rvalue reference.");
103
104 // Make sure it is not a pointer
105 static_assert(!std::is_pointer_v<T>, "T should't be a pointer");
106
107 // We need to clean-up the type for the pluginplayAny, basically we need to
108 // get it down to just cv qualifified.
109 using no_ref = std::remove_reference_t<T>;
110
111 m_is_cref_ = is_c_ref;
112 set_type_(typeid(no_ref)); // Sets type as seen by outside world
113 return add_type_check_<no_ref>(); // Set
114}

142template<typename T>
143type::any ModuleInput::wrap_value_(T&& new_value) const {
144 using clean_type = std::decay_t<T>;
145 using cref_type = const clean_type&;
146
147 // Did the user give us an AnyField already?
148 if constexpr(std::is_same_v<clean_type, type::any>) { return new_value; }
149
150 // Did the user give us a reference (so either T=U& or T=U&&)
151 constexpr bool is_ref = std::is_reference_v<T>;
152
153 // N.B. if is_ref is false, then ownership is being transferred to us. So
154 // even if we only need a reference we have to take the object by value or
155 // we'll end up with lifetime issues.
156 if(m_is_cref_ && is_ref)
157 return any::make_any_field<cref_type>(std::forward<T>(new_value));
158 else
159 return any::make_any_field<clean_type>(std::forward<T>(new_value));
160}
