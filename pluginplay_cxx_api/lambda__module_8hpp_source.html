189 LAMBDA_MOD_TYPE::LambdaModule(FxnType&& fxn) :
190  ModuleBase(this),
191  m_fxn_([&, da_fxn = std::forward<FxnType>(fxn)](
192  input_map inputs, const submodule_map&) -> result_map {
193  auto unwrapped_inputs = PropertyType::unwrap_inputs(std::move(inputs));
194 
195  using result_type = decltype(PropertyType::results());
196  using traits_type = typename result_type::traits_type;
197  using result_tuple = typename traits_type::tuple_of_fields;
198  constexpr auto n_results = traits_type::nfields;
199  result_tuple rv;
200 
201  if constexpr(n_results > 0) {
202  rv = std::move(std::apply(da_fxn, std::move(unwrapped_inputs)));
203  } else {
204  std::apply(da_fxn, std::move(unwrapped_inputs));
205  }
206 
207  auto is = std::make_index_sequence<n_results>();
208 
209  return wrap_results_(std::move(rv), is);
210  } // end lambda function
211  ) {
