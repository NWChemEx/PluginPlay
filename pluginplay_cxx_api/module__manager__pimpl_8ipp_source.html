31inline void ModuleManagerPIMPL::add_module(type::key key,
32 module_base_ptr base) {
33 assert_unique_key_(key);
34 auto uuid = utility::generate_uuid();
35 base->set_runtime(m_runtime_);
36 base->set_uuid(uuid);
37
38 cache::ModuleManagerCache::module_cache_pointer module_cache;
39 if(m_pcaches) {
40 auto internal_cache = m_pcaches->get_or_make_user_cache(uuid);
41 base->set_cache(internal_cache);
42 module_cache = m_pcaches->get_or_make_module_cache(key);
43 }
44
45 std::unique_ptr<ModulePIMPL> pimpl;
46 if(base->is_python()) {
47 // This is a hacky patch to allow multiple python modules to be
48 // added while avoiding the type_index collisions.
49 // TODO: remove when a more elegant solution is determined
50 m_py_bases[key] = base;
51 pimpl = std::make_unique<ModulePIMPL>(m_py_bases[key], module_cache);
52 } else {
53 std::type_index type(base->type());
54 if(!m_bases.count(type)) m_bases[type] = base;
55 pimpl = std::make_unique<ModulePIMPL>(m_bases[type], module_cache);
56 }
57 auto ptr = std::make_shared<Module>(std::move(pimpl));
58 ptr->set_name(key);
59 m_modules.emplace(std::move(key), ptr);
60}

71inline ModuleManagerPIMPL::shared_module ModuleManagerPIMPL::at(
72 const type::key& key) {
73 if(!count(key)) {
74 const std::string msg =
75 "ModuleManager has no module with key: '" + key + "'";
76 throw std::out_of_range(msg);
77 }
78 auto mod = m_modules.at(key);
79 // Loop over submodules filling them in from the defaults
80 for(auto& [k, v] : mod->submods()) {
81 const auto& type = v.type();
82 // Only change non-ready submodules
83 if(!v.ready() && m_defaults.count(type)) {
84 // Recursive to make sure that that module gets filled in
85 auto default_mod = at(m_defaults.at(type));
86 // Only change if the module is also ready
87 if(default_mod->ready(m_inputs.at(type)))
88 mod->change_submod(k, default_mod);
89 }
90 }
91 return mod;
92}

94inline bool ModuleManagerPIMPL::operator==(
95 const ModuleManagerPIMPL& rhs) const {
96 // Try to get out early
97 if(m_bases.size() != rhs.m_bases.size()) return false;
98 if(m_modules.size() != rhs.m_modules.size()) return false;
99 if(m_defaults.size() != rhs.m_defaults.size()) return false;
100
101 // TODO: Remove with the rest of the python hack
102 if(m_py_bases.size() != rhs.m_py_bases.size()) return false;
103 for(const auto& [k, v] : rhs.m_py_bases) {
104 if(!m_py_bases.count(k)) return false;
105 if(*m_py_bases.at(k) != *v) return false;
106 }
107
108 // Skip checking the values b/c implementations are compared by type
109 for(const auto& [k, v] : rhs.m_bases) {
110 if(!m_bases.count(k)) return false;
111 }
112
113 // Need to check the values b/c user may have switched options
114 for(const auto& [k, v] : rhs.m_modules) {
115 if(!m_modules.count(k)) return false;
116 if(*m_modules.at(k) != *v) return false;
117 }
118
119 // Easy since not pointers
120 if(m_defaults != rhs.m_defaults) return false;
121
122 return true;
123}
