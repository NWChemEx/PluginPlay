26TEMPLATE_PARAMS template<typename U, typename>
27ANY_FIELD_WRAPPER::AnyFieldWrapper(U&& value2wrap) :
28 base_type(wrap_value_(std::forward<U>(value2wrap))) {}

30TEMPLATE_PARAMS
31typename ANY_FIELD_WRAPPER::field_base_pointer ANY_FIELD_WRAPPER::clone_()
32 const {
33 using deep_copy_type = AnyFieldWrapper<clean_type>;
34 if constexpr(std::is_same_v<clean_type, python_value>) {
35 return std::make_unique<deep_copy_type>(as_python_wrapper());
36 } else {
37 const auto& val = base_type::template cast<const_ref_type>();
38 return std::make_unique<deep_copy_type>(val);
39 }
40}

42TEMPLATE_PARAMS
43bool ANY_FIELD_WRAPPER::are_equal_(const AnyFieldBase& rhs) const noexcept {
44 auto prhs = dynamic_cast<const my_type*>(&rhs);
45 if(prhs == nullptr) return false; // Not the same type
46
47 if constexpr(std::is_same_v<clean_type, python_value>) {
48 return as_python_wrapper() == rhs.as_python_wrapper();
49 } else {
50 const auto& lhs_value =
51 this->base_type::template cast<const_ref_type>();
52 const auto& rhs_value =
53 prhs->base_type::template cast<const_ref_type>();
54 return lhs_value == rhs_value;
55 }
56}
57
58TEMPLATE_PARAMS
59bool ANY_FIELD_WRAPPER::value_equal_(const AnyFieldBase& rhs) const noexcept {
60 // Regardless of how it's wrapped, every wrapped value can be converted to
61 // a const ref, so we first check rhs can be converted to the same const ref
62 if(!rhs.is_convertible<const_ref_type>()) return false;
63
64 const auto& lhs_value = this->base_type::template cast<const_ref_type>();
65 const auto& rhs_value = rhs.cast<const_ref_type>();
66 return lhs_value == rhs_value;
67}

69TEMPLATE_PARAMS
70std::ostream& ANY_FIELD_WRAPPER::print_(std::ostream& os) const {
71 using utilities::printing::operator<<;
72 if constexpr(utilities::type_traits::is_printable_v<T>) {
73 os << this->base_type::template cast<const T&>();
74 } else {
75 const auto* pvalue = &(this->base_type::template cast<const T&>());
76 os << "<" << typeid(T).name() << " " << pvalue << ">";
77 }
78 return os;
79}

81TEMPLATE_PARAMS
82typename ANY_FIELD_WRAPPER::python_value ANY_FIELD_WRAPPER::as_python_wrapper_()
83 const {
84 if(this->storing_python_object()) {
85 return this->base_type::template cast<python_value>();
86 }
87 const auto& my_value = this->base_type::template cast<const_ref_type>();
88 return python::make_python_wrapper(my_value);
89}

101TEMPLATE_PARAMS

107template<typename U>
108typename ANY_FIELD_WRAPPER::value_type ANY_FIELD_WRAPPER::wrap_value_(
