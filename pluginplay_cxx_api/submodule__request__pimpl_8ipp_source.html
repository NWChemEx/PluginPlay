48inline void SubmoduleRequestPIMPL::set_module(module_ptr ptr) {
49 if(!ptr) throw std::runtime_error("Pointer does not contain a module");
50 // Type will check that a property type was set
51 bool suitable = (ptr->property_types().count(type()) > 0);
52
53 // Work around for libc++ typeid inconsistency in Python
54 if(!suitable) {
55 for(const auto& property_type_info : ptr->property_types()) {
56 if(type().name() == property_type_info.name() ||
57 std::strcmp(type().name(), property_type_info.name()) == 0) {
58 suitable = true;
59 break;
60 }
61 }
62 }
63
64 if(!suitable) {
65 std::string msg("Module does not satisfy property type: ");
66 msg += utilities::printing::Demangler::demangle(type());
67 throw std::runtime_error(msg);
68 }
69
70 m_module_ = ptr;
71}
