28template<typename T>
29auto& ModuleResult::set_type() {
30 constexpr bool is_clean = std::is_same_v<std::decay_t<T>, T>;
31 static_assert(is_clean, "Results must be unqualified types.");
32 bounds_checking::TypeCheck<T> check;
33 auto l = [=](const type::any& value) { return check(value); };
34 set_type_check_(std::move(l));
35 return set_type_(typeid(T));
36}

38template<typename T>
39void ModuleResult::change(T&& new_value) {
40 using clean_T = std::decay_t<T>;
41 constexpr bool is_shared_any =
42 std::is_same_v<clean_T, shared_any> || // is shared_ptr<const any>
43 std::is_same_v<clean_T, std::shared_ptr<type::any>>; // no const
44 if constexpr(is_shared_any)
45 change_(new_value);
46 else if constexpr(detail_::IsCString<T>::value)
47 change_(wrap_value_(std::string(new_value)));
48 else
49 change_(wrap_value_(std::forward<T>(new_value)));
50}

52template<typename T>
53T ModuleResult::value() const {
54 using clean_T = std::decay_t<T>;
55 if constexpr(std::is_same_v<shared_any, clean_T>)
56 return at_();
57 else if constexpr(detail_::IsSharedPtr<clean_T>::value) {
58 using type = typename clean_T::element_type;
59 return T(at_(), &value<type&>());
60 } else
61 return any::any_cast<T>(*at_());
62}

64template<typename T>
65type::any ModuleResult::wrap_value_(T&& new_value) {
66 using clean_T = std::decay_t<T>;
67 // Is it already an AnyField
68 if constexpr(std::is_same_v<clean_T, type::any>) {
69 return std::forward<T>(new_value);
70 }
71 return any::make_any_field<clean_T>(std::forward<T>(new_value));
72}
