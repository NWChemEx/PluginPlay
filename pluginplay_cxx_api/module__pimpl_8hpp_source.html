40 inline auto time_stamp() {
41  using namespace std::chrono;
42  const auto now = system_clock::now();
43  const auto now_tt = system_clock::to_time_t(now);
44  const auto ms = duration_cast<milliseconds>(now.time_since_epoch()) % 1000;
45  std::stringstream ss;
46  ss << std::put_time(std::localtime(&now_tt), "%d-%m-%Y %H:%M:%S") << '.'
47  << std::setfill('0') << std::setw(3) << ms.count();
