581 inline bool ModuleBase::operator==(const ModuleBase& rhs) const noexcept {
582  if(type() != rhs.type()) return false;
583  if(has_description() != rhs.has_description()) return false;
584  if(has_description() && get_desc() != rhs.get_desc()) return false;
585  if(m_citations_ != rhs.m_citations_) return false;
586  if(m_is_python_ != rhs.m_is_python_) return false;
587  return std::tie(inputs(), results(), submods(), property_types()) ==
588  std::tie(rhs.inputs(), rhs.results(), rhs.submods(),
589  rhs.property_types());
