632 inline bool ModuleBase::operator==(const ModuleBase& rhs) const noexcept {
633  if(type() != rhs.type()) return false;
634  if(has_description() != rhs.has_description()) return false;
635  if(has_description() && get_desc() != rhs.get_desc()) return false;
636  if(m_citations_ != rhs.m_citations_) return false;
637  if(m_is_python_ != rhs.m_is_python_) return false;
638  return std::tie(inputs(), results(), submods(), property_types()) ==
639  std::tie(rhs.inputs(), rhs.results(), rhs.submods(),
640  rhs.property_types());
