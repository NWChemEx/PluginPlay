24inline bool ModuleInputPIMPL::is_valid(const type::any& new_value) const {
25 assert_type_set_();
26 for(const auto& [k, v] : m_checks_)

29}

38inline void ModuleInputPIMPL::set_value(type::any any) {
39 assert_type_set_();
40 if(!is_valid(any)) {
41 std::string msg("Input value: \"");
42 std::stringstream ss;
43 any.print(ss);
44 msg += ss.str();
45 msg += std::string("\" has failed bounds checks: ");
46 for(const auto& [x, y] : m_checks_) {
47 if(!y(any)) msg += x + " ";
48 }
49 throw std::invalid_argument(msg);
50 }
51 m_value_.swap(any);
52}

58inline void ModuleInputPIMPL::add_check(any_check check,
59 type::description desc) {
60 if(has_value())
61 if(!check(m_value_)) {
62 const auto msg = std::string("Value failed provided bounds "
63 "check: ") +
64 desc;
65 throw std::invalid_argument(desc);
66 }
67 m_checks_.emplace(std::move(desc), std::move(check));
68}

95inline bool operator==(const ModuleInputPIMPL& lhs,
96 const ModuleInputPIMPL& rhs) noexcept {
97 // Check boolean properties first
98 if(lhs.has_type() != rhs.has_type()) return false;
99 if(lhs.has_value() != rhs.has_value()) return false;
100 if(lhs.is_optional() != rhs.is_optional()) return false;
101 if(lhs.is_transparent() != rhs.is_transparent()) return false;
102 if(lhs.has_description() != rhs.has_description()) return false;
103
104 if(lhs.has_type() && (lhs.type() != rhs.type())) return false;
105 if(lhs.has_value() && (lhs.value() != rhs.value())) return false;
106 if(lhs.has_description() && (lhs.description() != rhs.description()))
107 return false;
108
109 return true;
110}
