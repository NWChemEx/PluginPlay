233 inline void ModuleResultPIMPL::set_value(shared_any new_value) {
234  // This'll throw bad optional_access if we haven't set the type
235  type();
236  bool is_corr_type = (*m_type_check_)(*new_value);
237  if(!is_corr_type) {

239  msg += utilities::printing::Demangler::demangle(type().name());
