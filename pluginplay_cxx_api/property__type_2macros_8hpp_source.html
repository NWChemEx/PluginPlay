153 .def("inputs", \

157 }) \
158 .def("results", \

161 return pluginplay::type::result_map(rvs.begin(), rvs.end()); \
162 }) \
163 .def("unwrap_inputs", \
164 [](prop_type& pt, pluginplay::type::input_map& inputs) { \
165 return pt.unwrap_inputs(inputs); \

168 [](prop_type& pt, pluginplay::type::input_map& inputs, \
169 pybind11::args args) { \

171 inputs, std::move(args)); \
172 }) \
173 .def("unwrap_results", \
