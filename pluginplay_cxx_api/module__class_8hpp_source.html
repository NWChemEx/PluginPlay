763 template<typename property_type, typename... Args>
764 auto Module::run_as(Args&&... args) {
765  check_property_type_(type::rtti{typeid(property_type)});
766  auto temp = inputs();
767  temp = property_type::wrap_inputs(temp, std::forward<Args>(args)...);
768  using r_type = decltype(property_type::unwrap_results(run(temp)));
769  using clean_t = std::decay_t<r_type>;
770  if constexpr(std::is_same_v<clean_t, void>) {
771  property_type::unwrap_results(run(temp));
772  } else {
773  auto rv = property_type::unwrap_results(run(temp));
774 
775  if constexpr(std::tuple_size_v<clean_t> == 1) {
776  return std::get<0>(rv);
777  } else {
778  return rv;
779  }
780  }
781 }
