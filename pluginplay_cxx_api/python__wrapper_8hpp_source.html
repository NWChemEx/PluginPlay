248template<typename T>
249bool PythonWrapper::is_convertible() noexcept {
250 // If we don't have a value we're not convertible
251 if(!has_value()) return false;
252
253 using clean_type = std::decay_t<T>;
254 // Does the user want a pybind11::object, or a reference to one?
255 if constexpr(std::is_same_v<clean_type, py_object_type>) {
256 return true;
257 }
258 // Does the user want a PythonWrapper, or a reference to one?
259 else if constexpr(std::is_same_v<clean_type, PythonWrapper>) {
260 return true;
261 }
262 // User wants us to try converting the Python object to a C++ object
263 else {
264 try {
265 if constexpr(std::is_copy_constructible_v<clean_type>) {
266 unwrap_().cast<clean_type>();
267 return true;
268 } else
269 return false;
270 } catch(...) { return false; }
271 }
272}

274template<typename T>
275bool PythonWrapper::is_convertible() const noexcept {
276 using clean_type = std::decay_t<T>;
277 constexpr bool to_mutable_ref = std::is_same_v<T, clean_type&>;
278
279 // Rule out mutable references
280 if constexpr(to_mutable_ref) { return false; }
281
282 // Now it's the same as non-const is_convertible
283 return const_cast<PythonWrapper*>(this)->is_convertible<T>();
284}

286template<typename T>
287T PythonWrapper::unwrap() {
288 if(!has_value())
289 throw std::runtime_error("Does not hold a value to unwrap");
290
291 using clean_type = std::decay_t<T>;
292 auto& py_value = unwrap_();
293
294 // Want a pybind11::object, pybind11::object&, or const pybind11::object&?
295 if constexpr(std::is_same_v<clean_type, py_object_type>) {
296 return py_value;
297 }
298 // Want a PythonWrapper, PythonWrapper&, or const PythonWrapper&?
299 else if constexpr(std::is_same_v<clean_type, PythonWrapper>) {
300 return *this;
301 }
302 // Want the Python object converted to a C++ value?
303 else if constexpr(std::is_same_v<clean_type, T>) {
304 return py_value.cast<clean_type>();
305 }
306 // User wants the converted C++ value by reference or const reference
307 else {
308 if(m_buffer_.empty() || m_buffer_.type() != typeid(clean_type)) {
309 auto cxx_value = py_value.cast<clean_type>();
310 m_buffer_ = std::move(cxx_value);
311 }
312 return boost::any_cast<T>(m_buffer_);
313 }
314}

316template<typename T>
317T PythonWrapper::unwrap() const {
318 using clean_type = std::decay_t<T>;
319 constexpr bool is_mutable_ref = std::is_same_v<T, clean_type&>;
320 static_assert(!is_mutable_ref, "Can not unwrap to mutable reference");
321
322 // okay to use non-const version since we know user doesn't want a mutable
323 // reference
324 return const_cast<PythonWrapper*>(this)->unwrap<T>();
325}
