21#define TPARAMS template<typename KeyType, typename ValueType>
22#define KEY_INJECTOR KeyInjector<KeyType, ValueType>
23
24TPARAMS
25KEY_INJECTOR::KeyInjector(key_key_type key_to_inject,
26 key_value_type value_to_inject,
27 sub_db_pointer sub_db) :
28 m_key_to_inject_(std::move(key_to_inject)),
29 m_value_to_inject_(std::move(value_to_inject)),
30 m_db_(sub_db) {
31 if(sub_db) return;
32 throw std::runtime_error("Expected a non-null database to wrap.");
33}
34
35TPARAMS
36typename KEY_INJECTOR::key_set_type KEY_INJECTOR::keys_() const {
37 auto rv = m_db_->keys();
38 for(auto& key : rv) key.erase(m_key_to_inject_);
39 return rv;
40}

42TPARAMS
43bool KEY_INJECTOR::count_(const_key_reference key) const noexcept {
44 // This actually can throw if there's not enough memory
45 try {
46 return m_db_->count(inject_(key));
47 } catch(...) {
48 std::cerr << "An uncaught exception was raised in KeyInjector::count_";
49 std::abort();
50 }
51}

53TPARAMS
54void KEY_INJECTOR::insert_(key_type key, mapped_type value) {
55 m_db_->insert(inject_(std::move(key)), std::move(value));
56}
57
58TPARAMS
59void KEY_INJECTOR::free_(const_key_reference key) { m_db_->free(inject_(key)); }
60
61TPARAMS
62typename KEY_INJECTOR::const_mapped_reference KEY_INJECTOR::at_(
63 const_key_reference key) const {
64 return m_db_->at(inject_(key));
65}

67TPARAMS
68typename KEY_INJECTOR::key_type KEY_INJECTOR::inject_(key_type key) const {
69 key.emplace(m_key_to_inject_, m_value_to_inject_);
70 return std::move(key);
71}
