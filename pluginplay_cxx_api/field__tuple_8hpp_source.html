330template<typename LHSType, typename RHSType, std::size_t depth = 0>
331auto union_guts(LHSType&& lhs, const RHSType& rhs) {
332 if constexpr(std::decay_t<RHSType>::size() == depth) { // End recursion
333 return lhs;
334 } else { // Non-empty RHS
335 auto itr = rhs.begin();
336 std::advance(itr, depth);
337 const auto& [k, v] = *itr;
338 using tuple_type = typename RHSType::traits_type::tuple_of_fields;
339 using v_type = std::tuple_element_t<depth, tuple_type>;
340 auto rv = lhs.template add_field<v_type>(k);
341 if constexpr(std::decay_t<LHSType>::is_input) {
342 if(v.has_value()) { rv.set_default(v.template value<v_type>()); }
343 }
344 return union_guts<decltype(rv), RHSType, depth + 1>(std::move(rv), rhs);
345 }
346}

390auto BUILDER_TYPE::add_field(type::key key) {
391 if(count(key))
392 throw std::invalid_argument("Key: " + key + " already exists.");
393 ElementType elem;
394 elem.template set_type<T>();
395 auto temp_values = add_field_(std::move(key), std::move(elem));

409template<typename T>
410void BUILDER_TYPE::set_default(T&& value) {
411 constexpr auto nfields = traits_type::nfields;
412 static_assert(nfields > 0, "Must have a field to set a default");
413 constexpr bool is_input = std::is_same_v<ElementType, ModuleInput>;
414 static_assert(is_input, "Returns cannot have a default value");
415 if constexpr(is_input) {
416 constexpr auto nm1 = nfields - 1;
417 std::get<nm1>(m_values_).second.set_default(std::forward<T>(value));
418 }
