330 template<typename LHSType, typename RHSType, std::size_t depth = 0>
331 auto union_guts(LHSType&& lhs, const RHSType& rhs) {
332  if constexpr(std::decay_t<RHSType>::size() == depth) { // End recursion
333  return lhs;
334  } else { // Non-empty RHS
335  auto itr = rhs.begin();
336  std::advance(itr, depth);
337  const auto& [k, v] = *itr;
338  using tuple_type = typename RHSType::traits_type::tuple_of_fields;
339  using v_type = std::tuple_element_t<depth, tuple_type>;
340  auto rv = lhs.template add_field<v_type>(k);
341  if constexpr(std::decay_t<LHSType>::is_input) {
342  if(v.has_value()) { rv.set_default(v.template value<v_type>()); }
343  }
344  return union_guts<decltype(rv), RHSType, depth + 1>(std::move(rv), rhs);
345  }
346 }
