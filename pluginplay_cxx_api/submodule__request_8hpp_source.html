496template<typename property_type, typename... Args>
497auto SubmoduleRequest::run_as(Args&&... args) {
498 if(type() != rtti_type(typeid(property_type)))
499 throw std::invalid_argument("Wrong property type");
500 return value().run_as<property_type>(std::forward<Args>(args)...);
